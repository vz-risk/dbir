### create our function to go from data to observed stat
uuid54fe730b <- function(df) {
    df %>%
        # calculate the count in each bin
        group_by(plus.dbir_year, `Security Researcher or Unrelated Third Party`) %>%
            summarize(x = sum(Misconfiguration, na.rm=TRUE)) %>%
        ungroup() %>%
        # calculate the sample size
        group_by(plus.dbir_year) %>%
            mutate(n = sum(x)) %>%
        ungroup() %>%
        # calculate the frequency and remove x and n as they are no longer needed
        mutate(freq = x/n) %>%
        select(-x, -n) %>%
        # calculate the chnage between years
        pivot_wider(names_from=plus.dbir_year, values_from=freq) %>%
        mutate(freq = `2020`-`2019`) %>%
        select(-`2019`, -`2020`) %>%
        # calculate the differences in change for the different levels of the independent variable
        pivot_wider(names_from=`Security Researcher or Unrelated Third Party`, values_from=freq) %>%
        summarize(diff_in_props = `TRUE` - `FALSE`) %>%
        pull(diff_in_props)
}

### create the correct records for our sample size.  We will simulate a null distribution out of this sample
